# dropdown

```js
import Dropdown from '@zcool/dropdown'
```

```.jsx
() => {
  function App() {
    const [isOpen, toggleOpen] = React.useState(false)

    // 子元素 click 时不关闭，需要返回 fasle
    function toggle() {
      toggleOpen(!isOpen)
      return false
    }

    return (
      <Dropdown text="选择水果" onToggle={toggle} >
        <div>
          <div>
            苹果
          </div>
          <div>
            香蕉
          </div>
          <div>
            <Dropdown text="更多" icon="none" isOpen={isOpen}>
              <div>
                <div>
                  猕猴桃
                </div>
                <div>
                  火龙果
                </div>
              </div>
            </Dropdown>
          </div>
        </div>
      </Dropdown>
    )
  }

  return <App />
}
```
