# Modal

```js
import Modal from '@zcool/modal'
```

```.jsx
() => {
  function App() {
    const [isOpen, toggleOpen] = React.useState(false)

    function toggle() {
      toggleOpen(!isOpen)
    }

    return (
      <>
        <Modal title="修改格式" isOpen={isOpen} onRequestClose={toggle}>
          this is a Modal
        </Modal>
        <Button onClick={toggle}>打开</Button>
      </>
    )
  }

  return <App />
}
```
