# Modal

```js
import Modal from '@zcool/modal'
```

```.jsx
() => {
  function App() {
    const [isOpen, toggleOpen] = React.useState(false)
    const [loading, setLoading] = React.useState(true)
    let timer = null

    React.useEffect(() => {
      timer = setTimeout(() => {
        setLoading(false)
      }, 10000)

      return function cleanUp() {
        clearTimeout(timer)
      }
    })

    function toggle() {
      toggleOpen(!isOpen)
    }


    return (
      <>
        <Modal
          title="修改格式"
          isOpen={isOpen}
          loading={loading}
          loadingPositionTop={10}
          onRequestClose={toggle}
          footer={
            <Button>
              知道了
            </Button>
          }
        >
          this is a Modal
        </Modal>
        <Button onClick={toggle}>打开</Button>
      </>
    )
  }

  return <App />
}
```
