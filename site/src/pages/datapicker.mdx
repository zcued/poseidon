# DataPicker

```js
import { DatePicker } from '@zcool/datepicker'
```

```.jsx
() => {
  function App() {
    function onChange(e) {
      console.log('点一个：',e)
    }

    return (
      <>
        <DatePicker placeholder="下载时间" onChange={onChange} />
        <RangePicker
          startPlaceholder="开始时间"
          endPlaceholder="结束时间"
          defaultValue={{
            from: '2019-01-01',
            to: '2019-01-31'
          }}
          onChange={onChange}
        />
      </>
    )
  }

  return <App />
}
```
