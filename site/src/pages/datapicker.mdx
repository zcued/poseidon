# DataPicker

```js
import { DatePicker } from '@zcool/datepicker'
```

```.jsx
() => {
  function App() {
    const [dateValue,setDateValue] = React.useState({
      from: {day: 1,month: 0,year: 2019},
      to: {day: 30,month: 0,year: 2019}
    })

    function onChange(e) {
      console.log('点一个：',e)
      setDateValue({...e})
    }

    return (
      <>
        <DatePicker
            placeholder="下载时间"
            onChange={onChange}
            defaultValue='2019-02-01'
        />
        <RangePicker
          startPlaceholder="开始时间"
          endPlaceholder="结束时间"
          onChange={onChange}
          value={dateValue}
        />
      </>
    )
  }

  return <App />
}
```
